<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-2xl sm:text-3xl font-semibold text-gray-900 dark:text-white mb-4">
    Quick Start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get your portfolio up and running in minutes
  </p>

  <Steps>
    <Step title="Clone and Install">
      Clone the repository and install dependencies:
      
      ```bash
      git clone https://github.com/Jorgemacias-12/Portfolio.git
      cd Portfolio
      bun install
      ```
    </Step>
    
    <Step title="Configure Your Content">
      Update the locale files in `src/locales/` with your personal information:
      
      ```json
      {
        "site": {
          "title": "Your Name | Software Developer",
          "description": "Your professional description"
        },
        "sections_data": {
          "about": {
            "greeting": "Hello, I am",
            "short_name": "Your Name",
            "position": "Your Position"
          }
        }
      }
      ```
    </Step>
    
    <Step title="Start Development Server">
      Run the development server to preview your portfolio:
      
      ```bash
      bun dev
      ```
      
      <Note>
        The site will be available at `http://localhost:4321`
      </Note>
    </Step>
    
    <Step title="Build and Deploy">
      Build your portfolio for production:
      
      ```bash
      bun build
      ```
      
      The optimized site will be in the `dist/` directory, ready for deployment.
    </Step>
  </Steps>
</div>
